#include <bits/stdc++.h>
using namespace std;

int n,m,in[10001],ans,vis[10001],a[10001];
vector<int>g[10001];

void TopoSort(){
    queue<int>s;
    ans=0;
    for(int i=1;i<=n;i++){
        if(!in[a[i]] && vis[a[i]])
            s.push(a[i]);
    }
    while(!s.empty()){
        int u = s.front();
        ans++;
        s.pop();
        for(auto v:g[u]){
            if(--in[v]==0&&vis[v])s.push(v);
        }
    }
    if(ans==n)cout<<"YES\n";
    else cout<<n-ans<<endl;
}

int main(){
    cin>>n;
    for(int i=1;i<=n;i++){

        for(int j=1;j<=m;j++){
            int y;
            cin>>y;
            g[a[i]].push_back(y);
            in[y]++;
        }
    }
    TopoSort();
    return 0;
}
