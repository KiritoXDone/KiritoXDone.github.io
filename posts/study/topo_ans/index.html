int n,m,c,in[N],vis[N],ans[N];

    queue<int>s;
    for(int i=1;i<=n;i++){
        if(!in[i]){
            s.push(i);
        }

    while(!s.empty()){
        int u = s.front();
        s.pop();
        if(vis[u])continue;
        vis[u]=1;
        for(auto [v,x]:g[u]){
            ans[v]=max(ans[v],ans[u]+x);
            if(--in[v]==0){
                s.push(v);
            }
        }
    }
    for(int i=1;i<=n;i++)cout<<ans[i]<<endl;
}
