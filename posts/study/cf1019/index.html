#include <bits/stdc++.h>
using namespace std;
#define ls now<<1
#define rs now<<1|1
#define endl "\n"
#define lowbit(x) ((x)&(-x))
typedef long long ll;
const int N=2e5+7, mod=1e9+7;

int n,k,a[N],s[N],ms[N];
bool flag;

void c1(){
    int l,r;
    int tmp=0;
    l=n+1,r=0;
    for(int i=1;i<=n;i++){
        tmp+=a[i];
        if(tmp>=0){
            l=i;
            break;
        }
    }
    tmp=0;
    for(int i=n;i>0;i--){
        tmp+=a[i];
        if(tmp>=0){
            r=i;
            break;
        }
    }
    if(l+1<r)flag=1;
}

void c2(){
    s[n] = ms[n] = a[n];
    for(int i=n-1; i>=1; i--){
          s[i] = s[i+1] + a[i];
          ms[i] = min(ms[i+1], s[i]);
    }
    int sum = 0;
    for(int i=1; i+2<=n; i++){
          sum += a[i];
          if(sum < 0) continue;
          if(s[i+1] >= ms[i+2]){
                flag = 1;
                return;
          }
    }
}



void solve(){
    cin>>n>>k;
    flag=0;
    for(int i=1;i<=n;i++){
        cin>>a[i];
        if(a[i]>k)a[i]=-1;
        else a[i]=1;
    }
    c1();
    c2();
    for(int i=1;i<n-i+1;i++)
        swap(a[i],a[n-i+1]);
    c2();
    if(flag)cout<<"Yes\n";
    else cout<<"No\n";
}

int main(){
    ios::sync_with_stdio(0);
    cin.tie(0);cout.tie(0);
    int t=1;
    cin>>t;
    while(t--)solve();
    return 0;
}

#include <bits/stdc++.h>
using namespace std;
#define ls now<<1
#define rs now<<1|1
#define endl "\n"
#define lowbit(x) ((x)&(-x))
typedef long long ll;
const int N=2e5+7, mod=1e9+7;

int n,a[N],ans[N];

void solve(){
    cin>>n;
    int mx=0;
    for(int i=1;i<=n;i++){
        cin>>a[i];
        ans[i]=0;
        mx=max(mx,a[i]);
    }
    int l=1,r=n;
    for(int i=1;i<=mx;i++){
        vector<int>b[2];
        int flag=0;
        for(int j=1;j<=n;j++){
            if(a[j]==-1){
                flag=1;
                continue;
            }
            if(a[j]==i)b[flag].push_back(j);
        }
        reverse(b[1].begin(),b[1].end());
        for(auto x:b[0]){
            if(i&1)ans[x]=r--;
            else ans[x]=l++;
        }
        for(auto x:b[1]){
            if(i&1)ans[x]=r--;
            else ans[x]=l++;
        }
    }
    for(int i=1;i<=n;i++){
        if(ans[i]==0)ans[i]=l;
        cout<<ans[i]<<" ";
    }
    cout<<endl;
}

int main(){
    ios::sync_with_stdio(0);
    cin.tie(0);cout.tie(0);
    int t=1;
    cin>>t;
    while(t--)solve();
    return 0;
