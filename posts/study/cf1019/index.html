A. Common Multiple#

不同数的个数

#include <bits/stdc++.h>
using namespace std;
#define ls now<<1
#define rs now<<1|1
#define endl "\n"
#define lowbit(x) ((x)&(-x))
typedef long long ll;
const int N=1e5+7, mod=1e9+7;

int n,a[N],ans;
map<int,int>vis;

void solve(){
    cin>>n;
    ans=0;
    for(int i=1;i<=n;i++){
        cin>>a[i];
        if(!vis[a[i]])ans++;
        vis[a[i]]++;
    }
    cout<<ans<<endl;
    vis.clear();
}

int main(){
    ios::sync_with_stdio(0);
    cin.tie(0);cout.tie(0);
    int t=1;
    cin>>t;
    while(t--)solve();
    return 0;
}

B. Binary Typewriter#

反转最多减少两次

#include <bits/stdc++.h>
using namespace std;
#define ls now<<1
#define rs now<<1|1
#define endl "\n"
#define lowbit(x) ((x)&(-x))
typedef long long ll;
const int N=1e5+7, mod=1e9+7;

int n;
string s;

void solve(){
    cin>>n>>s;
    int ans=0,now=0;
    for(int i=0;i<n;i++){
        if(s[i]-'0'!=now)ans+=2,now^=1;
        else ans++;
    }
    int flag=0;
    int pos=s.find("10");
    if(pos!=-1){
        ans--;
        pos=s.find("01",pos+1);
    }
    if(pos!=-1)ans--;
    cout<<ans<<endl;
}

int main(){
    ios::sync_with_stdio(0);
    cin.tie(0);cout.tie(0);
    int t=1;
    cin>>t;
    while(t--)solve();
    return 0;
}

C. Median Splits#

只分成三部分，所以只有三种情况：

  • 前后两部分满足；
  • 前面和中间满足；
  • 中间和后面满足。

只需要中位数小于等于 k 即可，那么大小关系只与 k 有关，如果 $a_i<=k$ 则 $a_i=1$ 反之，则 $a_i=-1$。

第一种情况，只需要正反各跑一遍看有没有两个不重叠的部分能使前缀和/后缀和大于等于 0 即可。

第二种可以用后缀和，统计后缀和当前值，后缀和最小值，如果后缀和当前值大于等于最小值，说明中间有部分是非负的。

最后一种可以将数组旋转，复用函数。

#include <bits/stdc++.h>
using namespace std;
#define ls now<<1
#define rs now<<1|1
#define endl "\n"
#define lowbit(x) ((x)&(-x))
typedef long long ll;
const int N=2e5+7, mod=1e9+7;

int n,k,a[N],s[N],ms[N];
bool flag;

void c1(){
    int l,r;
    int tmp=0;
    l=n+1,r=0;
    for(int i=1;i<=n;i++){
        tmp+=a[i];
        if(tmp>=0){
            l=i;
            break;
        }
    }
    tmp=0;
    for(int i=n;i>0;i--){
        tmp+=a[i];
        if(tmp>=0){
            r=i;
            break;
        }
    }
    if(l+1<r)flag=1;
}

void c2(){
    s[n] = ms[n] = a[n];
    for(int i=n-1; i>=1; i--){
          s[i] = s[i+1] + a[i];
          ms[i] = min(ms[i+1], s[i]);
    }
    int sum = 0;
    for(int i=1; i+2<=n; i++){
          sum += a[i];
          if(sum < 0) continue;
          if(s[i+1] >= ms[i+2]){
                flag = 1;
                return;
          }
    }
}



void solve(){
    cin>>n>>k;
    flag=0;
    for(int i=1;i<=n;i++){
        cin>>a[i];
        if(a[i]>k)a[i]=-1;
        else a[i]=1;
    }
    c1();
    c2();
    for(int i=1;i<n-i+1;i++)
        swap(a[i],a[n-i+1]);
    c2();
    if(flag)cout<<"Yes\n";
    else cout<<"No\n";
}

int main(){
    ios::sync_with_stdio(0);
    cin.tie(0);cout.tie(0);
    int t=1;
    cin>>t;
    while(t--)solve();
    return 0;
}

D. Local Construction#

注意次数，只有 log 次，所以按题意模拟也只会有 nlogn。

首先明确，删相对最大值时，删掉的一定是当前最大的值；同理，删相对最小值也都是当前最小的值。

按操作次数遍历，最终位置左侧和右侧的分开储存，为了避免目标位置的数被删掉，将某一侧储存的位置反转，即可使目标位置保留下来。

#include <bits/stdc++.h>
using namespace std;
#define ls now<<1
#define rs now<<1|1
#define endl "\n"
#define lowbit(x) ((x)&(-x))
typedef long long ll;
const int N=2e5+7, mod=1e9+7;

int n,a[N],ans[N];

void solve(){
    cin>>n;
    int mx=0;
    for(int i=1;i<=n;i++){
        cin>>a[i];
        ans[i]=0;
        mx=max(mx,a[i]);
    }
    int l=1,r=n;
    for(int i=1;i<=mx;i++){
        vector<int>b[2];
        int flag=0;
        for(int j=1;j<=n;j++){
            if(a[j]==-1){
                flag=1;
                continue;
            }
            if(a[j]==i)b[flag].push_back(j);
        }
        reverse(b[1].begin(),b[1].end());
        for(auto x:b[0]){
            if(i&1)ans[x]=r--;
            else ans[x]=l++;
        }
        for(auto x:b[1]){
            if(i&1)ans[x]=r--;
            else ans[x]=l++;
        }
    }
    for(int i=1;i<=n;i++){
        if(ans[i]==0)ans[i]=l;
        cout<<ans[i]<<" ";
    }
    cout<<endl;
}

int main(){
    ios::sync_with_stdio(0);
    cin.tie(0);cout.tie(0);
    int t=1;
    cin>>t;
    while(t--)solve();
    return 0;
}
