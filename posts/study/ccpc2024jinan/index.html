        dep[u].push_back(v);
        dep[v].push_back(u);
    }
    bool flag=0;
    int pos=1;
    for(int i=1;i<=n;i++){
        if((int)(dep[i].size())%2==0){
            flag=1,pos=i;
            break;
        }
    }
    if(!flag){

        return;
    }
    dfs(pos);

}
