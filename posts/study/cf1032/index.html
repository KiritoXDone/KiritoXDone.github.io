        for (int j = 1; j <= m; j++) {
            int tmp = row[i] + col[j];
            if (a[i][j] == mx) tmp--;
            if (tmp == cnt[mx]) {
                cout << ans - 1 << endl;
                return;
            }
        }
    }
   
    cout << ans << endl;
}

#define endl "\n"
typedef long long ll;

const int N = 1e5 + 7, mod = 1e9 + 7;

void solve() {
    ll n, s, x;
    cin >> n >> s >> x;
    vector<ll> a(n + 10), pre(n + 10);
    for (int i = 1; i <= n; i++) {
        cin >> a[i];
        pre[i] = pre[i - 1] + a[i];
    }

        map<ll, int> cnt;
        cnt[0] = 1;

            cnt[pre[i]]++;
        }
        return res;

}
