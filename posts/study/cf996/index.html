#include <bits/stdc++.h>
using namespace std;
#define ls now<<1
#define rs now<<1|1
#define endl "\n"
#define lowbit(x) ((x)&(-x))
typedef long long ll;
const int N=2e5+7, mod=1e9+7;

int n,k,l;
int a[N];

void solve(){
    cin>>n>>k>>l;
    k*=2,l*=2;
    for(int i=1;i<=n;i++){
        cin>>a[i];
        a[i]*=2;
    }
    int t=a[1];
    a[1]=0;
    for(int i=1;i<n;i++){
        if(a[i+1]-a[i]<k){
            if(a[i+1]+t>=a[i]+k)a[i+1]=a[i]+k;
            else a[i+1]+=t;
        }
        else{
            if(a[i+1]-t<=a[i]+k)a[i+1]=a[i]+k;
            else {
                int dt=((a[i+1]-t)-(a[i]+k))/2;
                t+=dt;
                a[i+1]=a[i]+dt+k;
            }
        }
    }
    if(a[n]+k<l)t+=l-(a[n]+k);
    cout<<t<<endl;
}

int main(){
    ios::sync_with_stdio(0);
    cin.tie(0);cout.tie(0);
    int t=1;
    cin>>t;
    while(t--)solve();
    return 0;
